c with STL学习笔记

发表于2023-02-24|更新于2023-03-19
|阅读量:

1.头文件：（1）万能头文件： #include<bits/stdc++.h>
（2）加快c++中的输入输出：#define FIO ios::sync_with_stdio(0);cin.tie(0); //不要忘记在main（）函数中写下：FIO;
（3）在c++中C语言程序加载库：#include < cstdio >
（4）#define 是一种预处理命令，用于定义宏，本质上是文本替换.
2.使用标准库空间std： using namespace std;
3.不要使用endl（复杂拖延时间），改用\n：(1)头文件替换： #define endl ‘\n’
（2）直接输出’\n’

vector

#include<bits/stdc++.h>  
using namespace std;
int main(){
    
    
    //初始化及遍历。 
    //vector<int> a(10,3);
    //vector<int> a(b.begin(),b.begin+3);
    int b[7]={1,2,3,4,5,6,7};
    vector<int> a(b,b+7);
    for(auto x:a){     
        cout<<x<<" ";
    }
    cout<<endl;
    
    //元素个数
    int sum;
    sum=a.size();     
    cout<<sum<<endl;
    
    //将数组大小改为3
    a.resize(3);      
    for(auto x:a){     
        cout<<x<<" ";
    }
    cout<<endl;
    
    //返回第一个和最后一个元素 
    int front=a.front();  
    int back=a.back();     
    cout<<front<<" "<<back<<endl;
    
    //清空a的元素
    a.clear();        
    
    //判断是否为空
    int z=a.empty();  
    cout<<z<<endl;
    
    //比较数组大小（从前往后比较）
    vector<int> bala(4,3),bili(3,4);
    cout<<"bala:";
    for(auto x:bala){     
        cout<<x<<" ";
    }
    cout<<endl<<"bili:";
    for(auto x:bili){     
        cout<<x<<" ";
    }
    cout<<endl;
    if(bala<bili)
        cout<<"bala<bili"<<endl;
    
    //在数组尾部插入和删除元素 
    a.push_back(5);
    a.push_back(6);
    a.push_back(7);
    a.push_back(8);
    for(auto x:a){     
        cout<<x<<" ";
    }
    cout<<endl;
    a.pop_back();
    for(auto x:a){     
        cout<<x<<" ";
    }
    cout<<endl;
    
    //a.begin();vector的第0个数
    //a.end();vector的最后一个的数的后面一个数
    //通常与for循环结合使用
    
    //倒置
    reverse(a.begin(),a.end());
    for(auto x:a){     
        cout<<x<<" ";
    }
    
    return 0;
}

pair

#include<bits/stdc++.h>
using namespace std;

int main()
{
    //初始化
    pair<string,int> p("hello",1);
    
    //调用第一个，第二个
    string a;
    int b;
    a=p.first;
    b=p.second;
    cout << a <<" "<<b<< endl;
    
    //嵌套
    pair<int,pair<int,int>> q(2,{1,3});
    int d;
    d=q.first;
    cout <<d<< endl;
    
    return 0;
}

string

#include<bits/stdc++.h>
using namespace std;

int main()
{
    //初始化
    string a= "abc";
    cout << a << endl;
    
    //添加
    a +='d';
    cout << a << endl;
    
    //输出
    cout<<a.substr(0)<<endl;//原字符串
    cout <<a.substr(1) << endl;//除去第一个字符的原字符串
    cout <<a.substr(0,10) << endl;//输出0到10的字符串
    
    //插入
    a.push_back('e');//尾插入
    a.insert(a.begin(),'1');//在begin()插入 
    cout << a<< endl;
    string b="456";
    a.insert(1,b,0,3);//在a的下标1字符处插入b的0到3的字符 
    cout << a<< endl;
    
    //判断
    cout <<a.empty()<< endl;
    
    //返回字母个数
    cout<<a.size()<<" "<<a.length()<<endl;
    
    //清空
    a.clear();
    cout<<a.empty()<<endl;
    
    return 0;
}

stack栈 

#include <bits/stdc++.h> 
using namespace std;

int main(){
    stack<int> s;
    s.push(3);
    s.push(4);
    s.push(5);
    cout<<s.size()<<endl;
    s.pop();     //弹出栈顶元素 
    cout<<s.top()<<endl;//返回栈顶元素 
    cout<<s.size()<<endl;
    return 0;
}

deque双向队列

#include <bits/stdc++.h> 
using namespace std;
int main()
{
    deque<int> ideq(20);//创建有n条数据的deque
    deque<int>::iterator pos;
    int i;
    for(int i=0;i<20;i++)
    {
        ideq[i]=i;
    }
    cout<<"输出deque中的数据:";
    for(int i=0;i<20;i++)
    {
        cout<<ideq[i]<<" ";
    }
    cout<<endl;
    //在头尾加入新的数据
    ideq.push_back(50);
    ideq.push_front(80);
    cout<<"输出deque中的数据:";
    for(pos=ideq.begin();pos!=ideq.end();pos++)
    {
        cout<<(*pos)<<" ";
    }cout<<endl;
    //查找
    int findx=15;
    cout<<"查找："<<findx<<endl;
    pos=find(ideq.begin(),ideq.end(),findx);
    if(pos!=ideq.end())
        cout<<"FIND "<<(*pos)<<" successfully"<<endl;
    else cout<<"NOT FIND"<<endl;
    //在头尾删除数据
    cout<<"..在头尾删除数据..waiting."<<endl;
    ideq.pop_back();//尾
    ideq.pop_front();//头
    cout<<endl;
    cout<<"请输出deque中的数据:";
    for(pos=ideq.begin();pos!=ideq.end();pos++)
    {
        cout<<(*pos)<<" ";
    }
    cout<<endl;
    return 0;
}

set 【集合】和 multiset
set不允许元素重复，如果有重复就会被忽略，但multiset允许.

#include <bits/stdc++.h> 
using namespace std;

int main(){
    set<int> s={1,2,3,4,5,6,7,9,8};
        cout<<s.size()<<endl;// 返回元素个数
        cout<<s.empty()<<endl; //返回set是否是空的
        //s.clear(); //清空
        s.begin(); //第0个数，支持++或--，返回前驱和后继
        s.end(); //最后一个的数的后面一个数，支持++或--，返回前驱和后继
        s.insert(8); //插入一个数
        s.find(4); //查找一个数
        cout<<s.count(5)<<endl; //返回某一个数的个数
        s.erase(8); //删除所以x  时间复杂度 O(k + logn)
        for(auto x:s){
        	cout<<x<<' ';
        }
    	s.erase(s.begin(),s.end());//删除一个迭代器
        for(auto x:s){
        	cout<<x<<' ';
        }
    return 0;
}
