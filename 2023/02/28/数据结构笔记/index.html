--- #线性表 ##顺序表

1.顺序表类型定义

typedef struct{
    ElemType data[];
    int length;
}SqList;

ElemType理解为数据类型，区别是其可以随时更改为需要的类型，与typedef搭配使用。

例如在该程序前加上语句

typedef int ElemType;
//则data[]为int类型；

2.顺序表的建立

void CreateList(SqLIst *&L,ElemType a[],int n){
int i;
L=(SqList *)malloc(sizeof(SqList));//为L分配空间
for(i=0;i<n;i++)
    L->data[i]=a[i];//输入L的值
L->length=n;    //L的长度为n
}

3.顺序表的初始化

//将L的存储空间的长度变为0即可
void InitList(SqList *&L){
L=(SqList *)malloc(sizeof(SqList));
L->length=0;}

4.顺序表的清空

void DestroyList(SqList *&L){
free(L);
}

5.判断是否为空

int ListEmpty(SqList *&L){
    return(L->length==0)
}

6.返回L的长度

int ListLength(SqList *&L){
    return(L->length)
}

7.输出L的值

void DispList(SqList *&L){
if(ListEmpty(L)) return;//先判断L是否为0
for(int i=0;i<L->length;i++)
    printf("%d",L->data[i]);
printf("\n");
}

8.获得L的第n个数据e

int GetElem(SqList *&L,int n,ElemType &e){
    if(n<1||n>L->length) return 0;//判断n是否为L数组的下标
e=L->data[n-1];//n是指第n个数，对应下标为n-1
return 1；
}

9.定位e位于L的位置

int LocateElem(SqList *L,ElemType e){
int n=0;
while(n<L->length && L->data[n]!=e) i++;
if(n>=L->length)
    return 0;
else
    return n+1;//L的下标从0开始计数，因此0为第一个数，因此返回n+1
}

10.插入e于L的第n位置

int ListInsert(SqList *&L,int n,ElemType e){
if(n<1 || n>L->length+1)
    return 0;
n--;    //将顺序表的为序转化为下标
for(int i=L->length;i>n;i--)//将n后的元素后移一位
    L->data[i]=L->data[i-1];
L->data[n]=e;
return 1;
}

11.删除第n个元素，获得删除的e

int ListDelete(SqList *&L,int n,ElemType &e){
if(n<1 || n>L->length+1)
    return 0;
n--;
e=L->data[n];
for(int i=n;i < L->length-1; i++)
    L->data[i]=L->data[i+1];
L->length--;
return 1;
}

12.合并LA，LB

int UnionList(SqList *LA,SqList *LB,SqList *LC){
    int i=0,j=0,l=0;
while(i<LA->length && j<LB->length){ //LA,LC依次比较插入LC中

    i++;}
    else{
    LC->data[k]=LB->data[j];

while(i<LA->length){  //LA中有剩余插入到LC中

##单链表
1.结构体

typedef int ElemType
typedef struct LNode{
    ElemType data;
    struct LNode *next;
}LinkNode;

2.头插法建立单链表

void CreateListF(LinkNode *&L,ElemType a[],int n){
LinkNode *s;
L=(LinkNode *)malloc(sizeof(LinkNode));
L->next=NULL;
for(int i=0;i<n;i++){
    s=(LinkNode *)malloc(sizeof(LinkNode));
    s->data=a[i];
    s->next=L->next;
    }
}

3.尾插法建立单链表

void CreateListR(LinkNode *&L,ElemType a[],int n){
LinkNode *s,*r;
L = (LinkNode*)malloc(sizeof(LinkNode));
L->next=NULL;
r=L;
for(int i=0;i<n;i++){
s = (LinkNode*)malloc(sizeof(LinkNode));
s->data=a[i];
r->next=s;
r=s;
    }
r->next=NULL;
}

4.初始化链表

void InitList(LinkNode *&L){
L=(LinkNode *)malloc(sizeof(LinkNode));
L->next=NULL;
}

5.清空链表

void DestroyList(LinkNode *&L){
LinkNode *pre=L,*p=pre->next;
while (p!=NULL){
    free(pre);
    pre=p;
    p=pre->next;
    }
free(pre);	//此时p为NULL,pre指向尾结点,释放它
}

6.判断是否为空

bool ListEmpty(LinkNode *L){
return(L->next==NULL);
}

7.返回链表的长度

int ListLength(LinkNode *L){

LinkNode *p=L;
int i=0;
while(p->next!=NULL){
    i++;
    p=p->next;
}
return i;
}

8.输出单链表

void DispList(LinkNode *L){
    LinkNode *p=L->next;
    while(p!=NULL){
        printf("%d ",p->data);
        p=p->next;
    }
    printf("\n");
    }

9.获得单链表的第n个数e

bool GetElem(LinkNode *L,int n,ElemType &e){
int i=0;
LinkNode *p=L;
if(n<=0) return false;
while(i<n && p!=NULL){
    i++;
    p=p->next;
}
if(p==NULL)
return false;
else{
    e=p->data;
    return true;
}
}

10.定位e位于单链表的位置

int LocateElem(LinkNode *L,ElemType e){
LinkNode *p=L->next;
int n=1;
while(p!=NULL && p->data!=e){
    n++;
    p=p->next;
}
if(p==NULL)
return 0;
else
return n;
}

11.插入一个结点

bool ListInsert(LinkNode *&L,int n,ElemType e){
LinkNode *p=L,*s;
int i=0;
while(p!=NULL && i<n-1){
    p=p->next;
    i++;
}
if(p==NULL || i>n-1)
return false;
s = (LinkNode*)malloc(sizeof(LinkNode));
s->data=e;
s->next=p->next;
p->next=s;
return true;
}

12.删除一个结点

bool ListDelete(LinkNode *&L,int n,ElemType &e){
    LinkNode *p = L, *q;
    int i = 0;
    while (p->next != NULL && i < n - 1){
        p = p->next;
        i++;
    	}
    if (p->next == NULL || i > n - 1)
        return false;
    q = p->next;
    e = q->data;
    p->next = q->next;
    free(q);
    return true;
}

13.main函数

int main(){
LinkNode *L1,*L2,L;
ElemType arr1[10]={2,4,6,8,10,12};
ElemType arr2[10]={1,3,5,7,9,11};
CreateListF(L1,arr1,6);

printf("得到L2的第%d个数e，e=%d\n",n,e);
printf("L2的%d位于第%d个\n",e,LocateElem(L2,7));

##双链表
1.结构体

typedef int ElemType;
typedef struct DNode{
    ElemType data;
    struct DNode *prior;
    struct DNode *next;
}DLinkNode;

2.头插法创建链表

void CreateListF(DLinkNode *&L,ElemType a[],int n){
L=(DLinkNode*)malloc(sizeof(DLinkNode));
L->prior=NULL;
L->next=NULL;
for(int i=0;i<n;i++){
    DLinkNode *s=(DLinkNode*)malloc(sizeof(DLinkNode));
    s->data=a[i];
    s->next=L->next;
    if(L->next!=NULL)
        L->next->prior=s;
    L->next=s;
    s->prior=L;
    }
}

3.尾插法创建链表

void CreateListR(DLinkNode *&L,ElemType a[],int n){
L=(DLinkNode *)malloc(sizeof(DLinkNode));
DLinkNode *r=L;
for(int i=0;i<n;i++){
    DLinkNode *s=(DLinkNode *)malloc(sizeof(DLinkNode));
    s->data=a[i];
    r->next=s;
    s->prior=L;
    r=s;
    }
r->next=NULL;
}

4.双链表的插入

bool ListInsert(DLinkNode *&L,int n,ElemType e){
DLinkNode *p=L;
int i=0;
while(p!=NULL || i<n-1 ){
    p=p->next;
    i++;
}
if(p==NULL || i>n-1)
    return false;
DLinkNode *s=(DLinkNode *)malloc(sizeof(DLinkNode));
s->data=e;
s->next=p->next;
s->prior=p;
if(p->next!=NULL)
p->next->prior=s;
p->next=s;
return true;
}

5.双链表的删除

bool ListDelete(DLinkNode *&L,int n,ElemType &e){
DLinkNode *p=L;
int i=0;
while(p!=NULL && i<n-1){
    p=p->next;
    i++;
}
if(p==NULL || i>n-1)
    return false;
DLinkNode *q=p->next;
if(q->next!=NULL)
q->next->prior=p;
p->next=q->next;
e=q->data;
free(q);
return true;
}

6.链表的输出

void DispList(DLinkNode *&L){
DLinkNode *p=L->next;
while(p!=NULL){
    printf("%d ",p->data);
    p=p->next;
}
printf("\n");
}

7.main函数
